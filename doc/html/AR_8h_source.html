110 void Newtonian_AW (double A[3], double &P, double dW[3], double &W, const double xjk[3], const double &mj, const double &mk, const double &mm2, const double &epi) {
111 
112  // distance
113  double rjk = std::sqrt(xjk[0]*xjk[0]+xjk[1]*xjk[1]+xjk[2]*xjk[2]);
114 
115  // mass parameters
116  double mmjk = mj*mk; // m_i*m_j
117  double wjk;
118  if (mm2>0 && epi>0) {
119  // Wjk = mm2 if m_i*m_j < epi*m'^2; 0 otherwise;
120  if (mmjk<epi*mm2) wjk = mm2;
121  else wjk = 0;
122  }
123  else {
124  wjk = mmjk; // Wjk = m_i*m_j
125  }
126 
127  P = mmjk / rjk; // Potential energy
128  W = wjk / rjk; // Transformation coefficient
129 
130  // Acceleration
131  double rjk3 = rjk*rjk*rjk;
132  double mor3 = mk / rjk3;
133  A[0] = mor3 * xjk[0];
134  A[1] = mor3 * xjk[1];
135  A[2] = mor3 * xjk[2];
136 
137  // dW/dr
138  mor3 = wjk / rjk3;
139  dW[0] = mor3 * xjk[0];
140  dW[1] = mor3 * xjk[1];
141  dW[2] = mor3 * xjk[2];

154 void Newtonian_Ap (double A[3], double &P, const double xi[3], const double xp[3], const double &mi, const double &mp){
